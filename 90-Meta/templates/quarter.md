# Timeline of Major Events

<%*
// 月名の配列
const monthNames = ["January", "February", "March", "April", "May", "June",
                    "July", "August", "September", "October", "November", "December"];

// 開始月から終了月までの月を生成
const months = [];
const start = parseInt(startMonth);
const end = parseInt(endMonth);

for (let m = start; m <= end; m++) {
  months.push({
    num: m,
    name: monthNames[m - 1]
  });
}

// 各月のセクションを生成
for (let i = 0; i < months.length; i++) {
  tR += `## Month ${i + 1} (${months[i].num}月 ${months[i].name})\n\n`;
  tR += `### 主要な出来事\n- \n- \n- \n\n`;
  tR += `### 目標進捗スナップショット\n`;
  tR += `- **修士研究**: %\n`;
  tR += `- **就職活動**: %\n`;
  tR += `- **スキル開発**: %\n`;
  tR += `- **健康・生活基盤**: %\n`;
  tR += `- **人間関係**: %\n`;
  tR += `- **趣味・財務**: %\n\n`;
  if (i < months.length - 1) tR += `---\n\n`;
}
