# Timeline & Quarterly Review

<%*
// 月名の配列
const monthNames = ["January", "February", "March", "April", "May", "June",
                    "July", "August", "September", "October", "November", "December"];

// 四半期とその月を生成
for (let q = 1; q <= 4; q++) {
  const startMonth = (q - 1) * 3 + 1;
  const endMonth = startMonth + 2;

  tR += `## Q${q} (${startMonth}-${endMonth}月)\n\n`;
  tR += `**四半期レビュー**: [[quarter-${year}-Q${q}]]\n\n`;
  tR += `### この四半期を一言で表すと\n**「」**\n\n`;
  tR += `### ハイライト\n`;
  tR += `- **進化したところ**: \n`;
  tR += `- **最も幸せを感じたこと**: \n`;
  tR += `- **最も心に残った一言**: 「」 - \n`;
  tR += `- **最も苦労したこと**: \n\n`;
  tR += `### 目標達成度サマリー\n`;
  tR += `- **修士研究**: % - \n`;
  tR += `- **就職活動**: % - \n`;
  tR += `- **スキル開発**: % - \n`;
  tR += `- **健康・生活基盤**: % - \n`;
  tR += `- **人間関係**: % - \n`;
  tR += `- **趣味・財務**: % - \n\n`;
  tR += `### 月次イベント\n`;

  for (let m = 0; m < 3; m++) {
    const monthNum = startMonth + m;
    const monthName = monthNames[monthNum - 1];
    tR += `#### ${monthNum}月 ${monthName}\n- \n- \n\n`;
  }

  if (q < 4) tR += `---\n\n`;
}
